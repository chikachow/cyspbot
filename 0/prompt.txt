export function flyTrustedIssuerForOrgSlug(orgSlug: string): TrustedOidcIssuer | null {
  if (!flyOrgSlugPattern.test(orgSlug)) {
    return null;
  }

  const issuer = `https://${flyOidcHost}/${orgSlug}`;

  return {
    allowedSigningAlgorithms: ["RS256"],
    issuer,
    jwksUri: new URL(`${issuer}/.well-known/jwks`),
  };
}

export function parseConfiguredFlyOrgSlugs(value: string): readonly string[] | null {
  const slugs = value
    .split(",")

  if (slugs.length === 0 || slugs.some((slug) => !flyOrgSlugPattern.test(slug))) {
    return null;
  }

  return slugs;
}
